def render_menu(&block)
  # ADDED ROLE ATTR TO COMPLY WITH ACCESSIBILITY RULES
  content_tag :ul, role: "list" do
    elements = block_given? ? [block.call(@view)] : []
    safe_join(elements + menu_items)
  end
end
