def acceptable_files
  return unless files.attached?
  files.each do |file|
    unless file.byte_size <= 50.megabyte
      errors.add(:files, "Maksymalny rozmiar pliku to 50MB")
    end
    acceptable_types = ["image/jpg", "image/jpeg", "image/gif", "image/png", "image/bmp", "application/pdf", "application/msword"]
    unless acceptable_types.include?(file.content_type)
      errors.add(:files, "Dozwolne rozszerzenia plik√≥w: jpg jpeg gif png bmp pdf doc")
    end
  end
end

  end
