def call
  if form.invalid?
    ap form.errors.full_messages
  end
  return broadcast(:invalid) if form.invalid?

  update_remark!
  broadcast(:ok)
end

def update_remark!
  Decidim.traceability.update!(
    remark,
    form.current_user,
    remark_attributes,
    visibility: "admin-only"
  )
end
