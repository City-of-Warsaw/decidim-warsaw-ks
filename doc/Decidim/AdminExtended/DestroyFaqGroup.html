def destroy_faq_group
  if faq_group.faqs.any?
    faq_group.faqs.each do |faq|
      Decidim.traceability.perform_action!(
        "delete",
        faq,
        current_user
      )
    end
  end

  Decidim.traceability.perform_action!(
    "delete",
    faq_group,
    current_user
  ) do
    faq_group.destroy!
  end
end
