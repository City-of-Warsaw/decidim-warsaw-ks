def create_page
  attributes = {
    title: @form.title,
    body: @form.body,
    gallery_id: @form.gallery_id,
    component: @form.current_component
  }

  @post = Decidim.traceability.create!(
    Decidim::Pages::Page,
    @current_user,
    attributes,
    visibility: "all"
  )
end
