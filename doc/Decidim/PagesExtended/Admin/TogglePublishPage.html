def call
  return broadcast(:invalid) if @page.invalid?

  if @page.published?
    unpublish_page
  else
    publish_page
    notify_followers!
  end
  broadcast(:ok)
end

def publish_page
  Decidim.traceability.perform_action!(
    :publish,
    @page,
    @current_user
  ) do
    @page.publish!
    @page
  end
end
