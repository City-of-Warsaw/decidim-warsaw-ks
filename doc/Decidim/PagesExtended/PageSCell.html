def description
  text = if model.is_a?(Decidim::StaticPage)
           model.content
         elsif model.is_a?(Decidim::Pages::Page)
           model.body
         end
  processed_text = translated_attribute(text)
  decidim_sanitize(truncate(strip_tags(processed_text), length: 210))
end
