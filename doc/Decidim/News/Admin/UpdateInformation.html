def call
  return broadcast(:invalid) if form.invalid?

  update_information!
  add_gallery(information)
  send_notification if information.published?

  broadcast(:ok, information)
end

def update_information!
  Decidim.traceability.update!(
    information,
    form.current_user,
    attributes,
    visibility: 'admin-only'
  )
end
