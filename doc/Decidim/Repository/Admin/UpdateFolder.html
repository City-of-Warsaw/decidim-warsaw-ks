def call
  return broadcast(:invalid) if form.invalid?

  update_folder!

  broadcast(:ok, folder)
end

def update_folder!
  Decidim.traceability.update!(
    folder,
    current_user,
    folder_params,
    visibility: "admin-only"
  )
end
