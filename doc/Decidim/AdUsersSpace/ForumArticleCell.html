def card_size
  if options[:size] == :search
    "decidim/ad_users_space/forum_article_search"
  elsif options[:size] == :s
    "decidim/ad_users_space/forum_article_s"
  else
    "decidim/ad_users_space/forum_article_m"
  end
end
